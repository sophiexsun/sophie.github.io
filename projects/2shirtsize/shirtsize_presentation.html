#Step5: Create Random Forest in the specified number of trees

#For Women
for num in n_trees:
    print(num)
    rf_f = RandomForestClassifier(n_estimators=num,
                                oob_score=True,

                                random_state=2020,
                                class_weight='balanced',

    rf_f.fit(Xf, yf)
